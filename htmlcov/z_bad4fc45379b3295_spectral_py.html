26def median(x: Float[Tensor, "... size"], axis: int) -> Float[Tensor, "..."]: 

27 """ 

28 Implements a median calculation that matches numpy's 

29 behavior for an even number of elements and includes 

30 the same bias correction used by scipy's implementation. 

31 see https://github.com/scipy/scipy/blob/main/scipy/signal/_spectral_py.py#L2066 

32 """ # noqa: E501 

33 n = x.shape[axis] 

34 ii_2 = 2 * torch.arange(1.0, (n - 1) // 2 + 1) 

35 bias = 1 + torch.sum(1.0 / (ii_2 + 1) - 1.0 / ii_2) 

36 return torch.quantile(x, q=0.5, axis=axis) / bias

39def _validate_shapes( 

40 x: Tensor, nperseg: int, y: Optional[Tensor] = None 

41) -> None: 

42 if x.shape[-1] < nperseg: 

43 raise ValueError( 

44 "Number of samples {} in input x is insufficient " 

45 "for number of fft samples {}".format(x.shape[-1], nperseg) 

46 ) 

47 elif x.ndim > 3: 

48 raise ValueError( 

49 f"Can't compute spectral density on tensor with shape {x.shape}" 

50 ) 

51 

52 if y is None: 

53 return 

54 

55 # acceptable combinations of shapes: 

56 # x: time, y: time 

57 # x: channel x time, y: time OR channel x time 

58 # x: batch x channel x time, y: batch x channel x time OR batch x time 

59 if x.shape[-1] != y.shape[-1]: 

60 raise ValueError( 

61 "Time dimensions of x and y tensors must " 

62 "be the same, found {} and {}".format(x.shape[-1], y.shape[-1]) 

63 ) 

64 elif x.ndim == 1 and not y.ndim == 1: 

65 raise ValueError( 

66 "Can't compute cross spectral density of " 

67 "1D tensor x with {}D tensor y".format(y.ndim) 

68 ) 

69 elif x.ndim > 1 and y.ndim == x.ndim: 

70 if not y.shape == x.shape: 

71 raise ValueError( 

72 "If x and y tensors have the same number " 

73 "of dimensions, shapes must fully match. " 

74 "Found shapes {} and {}".format(x.shape, y.shape) 

75 ) 

76 elif x.ndim > 1 and y.ndim != (x.ndim - 1): 

77 raise ValueError( 

78 "Can't compute cross spectral density of " 

79 "tensors with shapes {} and {}".format(x.shape, y.shape) 

80 ) 

81 elif x.ndim > 2 and y.shape[0] != x.shape[0]: 

82 raise ValueError( 

83 "If x is a 3D tensor and y is a 2D tensor, " 

84 "0th batch dimensions must match, but found " 

85 "values {} and {}".format(x.shape[0], y.shape[0]) 

86 )

89def fast_spectral_density( 

90 x: TimeSeries1to3d, 

91 nperseg: int, 

92 nstride: int, 

93 window: Float[Tensor, " {nperseg//2+1}"], 

94 scale: float, 

95 average: str = "median", 

96 y: Optional[TimeSeries1to3d] = None, 

97) -> FrequencySeries1to3d: 

98 """ 

99 Compute the power spectral density of a multichannel 

100 timeseries or a batch of multichannel timeseries, or 

101 the cross power spectral density of two such timeseries. 

102 This implementation is non-exact for the two lowest 

103 frequency bins, since it implements centering using the 

104 mean for the entire timeseries rather than on a per- 

105 window basis. The benefit of this is a faster implementation, 

106 which might be beneficial for cases where the lowest 

107 frequency components are going to be discarded anyway. 

108 

109 Args: 

110 x: 

111 The timeseries tensor whose power spectral density 

112 to compute, or for cross spectral density the 

113 timeseries whose fft will be conjugated. Can have 

114 shape `(batch_size, num_channels, length * sample_rate)`, 

115 `(num_channels, length * sample_rate)`, or 

116 `(length * sample_rate)`. 

117 nperseg: 

118 Number of samples included in each FFT window 

119 nstride: 

120 Stride between FFT windows 

121 window: 

122 Window array to multiply by each FFT window before 

123 FFT computation. Should have length `nperseg // 2 + 1`. 

124 scale: 

125 Scale factor to multiply the FFT'd data by, related to 

126 desired units for output tensor (e.g. letting this equal 

127 `1 / (sample_rate * (window**2).sum())` will give output 

128 units of density, $\\text{Hz}^-1$$. 

129 average: 

130 How to aggregate the contributions of each FFT window to 

131 the spectral density. Allowed options are `'mean'` and 

132 `'median'`. 

133 y: 

134 Timeseries tensor to compute cross spectral density 

135 with `x`. If left as `None`, `x`'s power spectral 

136 density will be returned. Otherwise, if `x` is 1D, 

137 `y` must also be 1D. If `x` is 2D, the assumption 

138 is that this represents a single multi-channel timeseries, 

139 and `y` must be either 2D or 1D. In the former case, 

140 the cross-spectral densities of each channel will be 

141 computed individually, so `y` must have the same shape as `x`. 

142 Otherwise, this will compute the CSD of each of `x`'s channels 

143 with `y`. If `x` is 3D, this will be assumed to be a batch 

144 of multi-channel timeseries. In this case, `y` can either 

145 be 3D, in which case each channel of each batch element will 

146 have its CSD calculated or 2D, which has two different options. 

147 If `y`'s 0th dimension matches `x`'s 0th dimension, it will 

148 be assumed that `y` represents a batch of 1D timeseries, and 

149 for each batch element this timeseries will have its CSD with 

150 each channel of the corresponding batch element of `x` 

151 calculated. Otherwise, it sill be assumed that `y` represents 

152 a single multi-channel timeseries, in which case each channel 

153 of `y` will have its CSD calculated with the corresponding 

154 channel in `x` across _all_ of `x`'s batch elements. 

155 Returns: 

156 Tensor of power spectral densities of `x` or its cross spectral 

157 density with the timeseries in `y`. 

158 """ 

159 

160 _validate_shapes(x, nperseg, y) 

161 

162 if x.ndim > 2: 

163 # stft only works on 2D input, so roll the 

164 # channel dimension out along the batch 

165 batch_size = x.shape[0] 

166 num_channels = x.shape[1] 

167 x = x.reshape(-1, x.shape[-1]) 

168 

169 if y is not None and y.ndim == 3: 

170 # do the same thing for y if necessary 

171 y = y.reshape(-1, x.shape[-1]) 

172 else: 

173 # this is just a multichannel timeseries, so 

174 # we can ignore batch reshapes later 

175 batch_size = None 

176 

177 x = x - x.mean(axis=-1, keepdims=True) 

178 fft = torch.stft( 

179 x, 

180 n_fft=nperseg, 

181 hop_length=nstride, 

182 window=window, 

183 normalized=False, 

184 center=False, 

185 return_complex=True, 

186 ) 

187 

188 other = fft 

189 if y is not None: 

190 y = y - y.mean(axis=-1, keepdims=True) 

191 y_fft = torch.stft( 

192 y, 

193 n_fft=nperseg, 

194 hop_length=nstride, 

195 window=window, 

196 normalized=False, 

197 center=False, 

198 return_complex=True, 

199 ) 

200 if batch_size is not None and fft.shape[0] != y_fft.shape[0]: 

201 # x is batched but y is not batched, so expand x's 0th 

202 # dimension back out so that we can multiply these arrays 

203 nfreq = nperseg // 2 + 1 

204 fft = fft.reshape(batch_size, num_channels, nfreq, -1) 

205 y_fft = y_fft[:, None] 

206 other = y_fft 

207 

208 fft = torch.conj(fft) * other 

209 if y is None: 

210 fft = fft.real 

211 

212 # some overly complex logic to find the frequency dimension 

213 # to apply this inscrutable signal processing step of 

214 # multiplying the non-edge frequency bins by 2 

215 # TODO: move this after aggregation since it won't affect 

216 # output value, and we have one less axis to do it over? 

217 stop = None if nperseg % 2 else -1 

218 if x.ndim == 1: 

219 fft[1:stop] *= 2 

220 elif fft.ndim < 4: 

221 fft[:, 1:stop] *= 2 

222 else: 

223 fft[:, :, 1:stop] *= 2 

224 fft *= scale 

225 

226 if average == "mean": 

227 fft = fft.mean(axis=-1) 

228 else: 

229 # if this is a cross spectral density, fft will 

230 # be complex and so we'll compute the median along 

231 # both axes 

232 if y is not None: 

233 real_median = median(fft.real, -1) 

234 imag_median = 1j * median(fft.imag, -1) 

235 fft = real_median + imag_median 

236 else: 

237 fft = median(fft, -1) 

238 

239 if fft.ndim == 2 and batch_size is not None: 

240 # if we still haven't expanded the batch dimension 

241 # back out, do so now 

242 fft = fft.reshape(batch_size, num_channels, -1) 

243 return fft

246def spectral_density( 

247 x: TimeSeries1to3d, 

248 nperseg: int, 

249 nstride: int, 

250 window: Float[Tensor, " {nperseg//2+1}"], 

251 scale: float, 

252 average: str = "median", 

253) -> FrequencySeries1to3d: 

254 """ 

255 Compute the power spectral density of a multichannel 

256 timeseries or a batch of multichannel timeseries. 

257 This implementation is exact for all frequency bins, but 

258 slower than the fast implementation. 

259 

260 Args: 

261 x: 

262 The timeseries tensor whose power spectral density 

263 to compute, or for cross spectral density the 

264 timeseries whose fft will be conjugated. Can have 

265 shape `(batch_size, num_channels, length * sample_rate)`, 

266 `(num_channels, length * sample_rate)`, or 

267 `(length * sample_rate)`. 

268 nperseg: 

269 Number of samples included in each FFT window 

270 nstride: 

271 Stride between FFT windows 

272 window: 

273 Window array to multiply by each FFT window before 

274 FFT computation. Should have length `nperseg // 2 + 1`. 

275 scale: 

276 Scale factor to multiply the FFT'd data by, related to 

277 desired units for output tensor (e.g. letting this equal 

278 `1 / (sample_rate * (window**2).sum())` will give output 

279 units of density, $\\text{Hz}^-1$$. 

280 average: 

281 How to aggregate the contributions of each FFT window to 

282 the spectral density. Allowed options are `'mean'` and 

283 `'median'`. 

284 """ 

285 

286 _validate_shapes(x, nperseg) 

287 

288 # for non-fast implementation, we need to unfold 

289 # the tensor along the time dimension ourselves 

290 # to detrend each segment individually, so start 

291 # by converting x to a 4D tensor so we can use 

292 # torch's Unfold op 

293 if x.ndim == 1: 

294 reshape = [] 

295 x = x[None, None, None, :] 

296 elif x.ndim == 2: 

297 reshape = [len(x)] 

298 x = x[None, :, None, :] 

299 elif x.ndim == 3: 

300 reshape = list(x.shape[:-1]) 

301 x = x[:, :, None, :] 

302 

303 # calculate the number of segments and trim x along 

304 # the time dimensions so that we can unfold it exactly 

305 num_segments = (x.shape[-1] - nperseg) // nstride + 1 

306 stop = (num_segments - 1) * nstride + nperseg 

307 x = x[..., :stop] 

308 

309 # unfold x into overlapping segments and detrend and window 

310 # each one individually before computing the rfft. Unfold 

311 # will produce a batch x (num_channels * num_segments) x nperseg 

312 # shaped tensor 

313 unfold_op = torch.nn.Unfold((1, num_segments), dilation=(1, nstride)) 

314 x = unfold_op(x) 

315 x = x - x.mean(axis=-1, keepdims=True) 

316 x *= window 

317 

318 # after the fft, we'll have a 

319 # batch x (num_channels * num_segments) x nfreq 

320 # sized tensor 

321 fft = torch.fft.rfft(x, axis=-1).abs() ** 2 

322 

323 if nperseg % 2: 

324 fft[..., 1:] *= 2 

325 else: 

326 fft[..., 1:-1] *= 2 

327 fft *= scale 

328 

329 # unfold the batch and channel dimensions back 

330 # out if there were any to begin with, putting 

331 # the segment dimension as the second to last 

332 reshape += [num_segments, -1] 

333 fft = fft.reshape(*reshape) 

334 

335 if average == "mean": 

336 return fft.mean(axis=-2) 

337 else: 

338 return median(fft, -2)

341def truncate_inverse_power_spectrum( 

342 psd: PSDTensor, 

343 fduration: Union[Float[Tensor, " time"], float], 

344 sample_rate: float, 

345 highpass: Optional[float] = None, 

346 lowpass: Optional[float] = None, 

347) -> PSDTensor: 

348 """ 

349 Truncate the length of the time domain response 

350 of a whitening filter built using the specified 

351 `psd` so that it has maximum length `fduration` 

352 seconds. This is meant to mitigate the impact 

353 of sharp features in the background PSD causing 

354 time domain responses longer than the segments 

355 to which the whitening filter will be applied. 

356 

357 Implementation details adapted from 

358 https://github.com/vivinousi/gw-detection-deep-learning/blob/203966cc2ee47c32c292be000fb009a16824b7d9/modules/whiten.py#L8 

359 

360 Args: 

361 psd: 

362 The one-sided power spectraul density used 

363 to construct a whitening filter. 

364 fduration: 

365 Desired length in seconds of the time domain 

366 response of a whitening filter built using 

367 this PSD, or a window of this length to taper 

368 the edges of the time domain response of the 

369 filter. If passed as a float, a Hann window 

370 of this length will be used. 

371 sample_rate: 

372 Rate at which the time domain data to which 

373 the whitening filter will be applied has been 

374 sampled. 

375 highpass: 

376 If specified, will zero out the frequency response 

377 of all frequencies below this value in Hz. If left 

378 as `None`, no highpass filtering will be applied. 

379 lowpass: 

380 If specified, will zero out the frequency response 

381 of all frequencies above this value in Hz. If left 

382 as `None`, no lowpass filtering will be applied. 

383 Returns: 

384 The PSD with its time domain response truncated 

385 to `fduration` and any filtered frequencies 

386 tapered. 

387 """ # noqa: E501 

388 

389 num_freqs = psd.size(-1) 

390 N = (num_freqs - 1) * 2 

391 

392 # use the inverse of the ASD as the 

393 # impulse response function 

394 inv_asd = 1 / psd**0.5 

395 

396 # zero out frequencies if we want the filter 

397 # to perform highpass/lowpass filtering 

398 df = sample_rate / N 

399 if highpass is not None: 

400 idx = int(highpass / df) 

401 inv_asd[:, :, :idx] = 0 

402 if lowpass is not None: 

403 idx = int(lowpass / df) 

404 inv_asd[:, :, idx:] = 0 

405 

406 if inv_asd.size(-1) % 2: 

407 inv_asd[:, :, -1] = 0 

408 

409 # now convert to time domain representation 

410 q = torch.fft.irfft(inv_asd, n=N, norm="forward", dim=-1) 

411 

412 # taper the edges of the TD filter 

413 if isinstance(fduration, Tensor): 

414 pad = fduration.size(-1) // 2 

415 window = fduration 

416 else: 

417 pad = int(fduration * sample_rate / 2) 

418 window = torch.hann_window(2 * pad, dtype=torch.float64) 

419 window = window.to(q.device) 

420 

421 # 0 out anything else between the tapering regions 

422 q[:, :, :pad] *= window[-pad:] 

423 q[:, :, -pad:] *= window[:pad] 

424 if 2 * pad < q.size(-1): 

425 q[:, :, pad : q.size(-1) - pad] = 0 

426 

427 # convert back to the frequency domain 

428 # to build the desired PSD 

429 inv_asd = torch.fft.rfft(q, n=N, norm="forward", dim=-1) 

430 inv_psd = inv_asd * inv_asd.conj() 

431 psd = 1 / inv_psd.abs() 

432 return psd / 2

435def normalize_by_psd( 

436 X: WaveformTensor, 

437 psd: PSDTensor, 

438 sample_rate: float, 

439 pad: int, 

440): 

441 # compute the FFT of the section we want to whiten 

442 # and divide it by the ASD of the background section. 

443 # If the ASD of any background bin hit inf, set the 

444 # corresponding bin to 0 

445 X = X - X.mean(-1, keepdims=True) 

446 X_tilde = torch.fft.rfft(X.double(), norm="forward", dim=-1) 

447 X_tilde = X_tilde / psd**0.5 

448 X_tilde[torch.isnan(X_tilde)] = 0 

449 

450 # convert back to the time domain and normalize 

451 # TODO: what's this normalization factor? 

452 X = torch.fft.irfft(X_tilde, n=X.shape[-1], norm="forward", dim=-1) 

453 X = X.float() / sample_rate**0.5 

454 

455 # slice off corrupted data at edges of kernel 

456 X = X[:, :, pad:-pad] 

457 return X

460def whiten( 

461 X: WaveformTensor, 

462 psd: PSDTensor, 

463 fduration: Union[Float[Tensor, " time"], float], 

464 sample_rate: float, 

465 highpass: Optional[float] = None, 

466 lowpass: Optional[float] = None, 

467) -> WaveformTensor: 

468 """ 

469 Whiten a batch of timeseries using the specified 

470 background one-sided power spectral densities (PSDs), 

471 modified to have the desired time domain response length 

472 `fduration` and possibly to highpass/lowpass filter. 

473 

474 Args: 

475 X: 

476 batch of multichannel timeseries to whiten 

477 psd: 

478 PSDs use to whiten the data. The frequency 

479 response of the whitening filter will be roughly 

480 the inverse of the square root of this PSD, ensuring 

481 that data from the same distribution will have 

482 approximately uniform power after whitening. 

483 If 2D, each batch element in `X` will be whitened 

484 using the same PSDs. If 3D, each batch element will 

485 be whitened by the PSDs contained along the 0th 

486 dimenion of `psd`, and so the first two dimensions 

487 of `X` and `psd` should match. 

488 fduration: 

489 Desired length in seconds of the time domain 

490 response of a whitening filter built using 

491 this PSD, or a window of this length to taper 

492 the edges of the time domain response of the 

493 filter. If passed as a float, a Hann window 

494 of this length will be used. Moreover, half of 

495 this length will be removed from each edge of 

496 the whitened timeseries to account for filter 

497 settle-in time. 

498 sample_rate: 

499 Rate at which the data in `X` has been sampled 

500 highpass: 

501 The frequency in Hz at which to highpass filter 

502 the data, setting the frequency response in the 

503 whitening filter to 0. If left as `None`, no 

504 highpass filtering will be applied. 

505 lowpass: 

506 The frequency in Hz at which to lowpass filter 

507 the data, setting the frequency response in the 

508 whitening filter to 0. If left as `None`, no 

509 lowpass filtering will be applied. 

510 Returns: 

511 Batch of whitened multichannel timeseries with 

512 `fduration / 2` seconds trimmed from each side. 

513 """ 

514 

515 # figure out how much data we'll need to slice 

516 # off after whitening 

517 if isinstance(fduration, Tensor): 

518 pad = fduration.size(-1) // 2 

519 else: 

520 pad = int(fduration * sample_rate / 2) 

521 

522 N = X.size(-1) 

523 if N <= (2 * pad): 

524 raise ValueError( 

525 ( 

526 "Not enough timeseries samples {} for number of " 

527 "padded samples {}" 

528 ).format(N, 2 * pad) 

529 ) 

530 

531 # normalize the number of expected dimensions in the PSD 

532 while psd.ndim < 3: 

533 psd = psd[None] 

534 

535 # possibly interpolate our PSD to match the number 

536 # of frequency bins we expect to get from X 

537 num_freqs = N // 2 + 1 

538 if psd.size(-1) != num_freqs: 

539 # TODO: does there need to be any rescaling to 

540 # keep the integral of the PSD constant? 

541 psd = torch.nn.functional.interpolate( 

542 psd, size=(num_freqs,), mode="linear" 

543 ) 

544 

545 # truncate it to have the desired 

546 # time domain response length 

547 psd = truncate_inverse_power_spectrum( 

548 psd, 

549 fduration, 

550 sample_rate, 

551 highpass, 

552 lowpass, 

553 ) 

554 

555 return normalize_by_psd(X, psd, sample_rate, pad)
