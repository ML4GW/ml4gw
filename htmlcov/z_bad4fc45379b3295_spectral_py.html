27def median(x: Float[Tensor, "... size"], axis: int) -> Float[Tensor, "..."]: 

28 """ 

29 Implements a median calculation that matches numpy's 

30 behavior for an even number of elements and includes 

31 the same bias correction used by 

32 `scipy's implementation <https://github.com/scipy/scipy/blob/main/scipy/signal/_spectral_py.py#L2066>`_. 

33 """ # noqa: E501 

34 n = x.shape[axis] 

35 ii_2 = 2 * torch.arange(1.0, (n - 1) // 2 + 1) 

36 bias = 1 + torch.sum(1.0 / (ii_2 + 1) - 1.0 / ii_2) 

37 return torch.quantile(x, q=0.5, axis=axis) / bias

40def _validate_shapes(x: Tensor, nperseg: int, y: Tensor | None = None) -> None: 

41 if x.shape[-1] < nperseg: 

42 raise ValueError( 

43 f"Number of samples {x.shape[-1]} in input x is insufficient " 

44 f"for number of fft samples {nperseg}" 

45 ) 

46 elif x.ndim > 3: 

47 raise ValueError( 

48 f"Can't compute spectral density on tensor with shape {x.shape}" 

49 ) 

50 

51 if y is None: 

52 return 

53 

54 # acceptable combinations of shapes: 

55 # x: time, y: time 

56 # x: channel x time, y: time OR channel x time 

57 # x: batch x channel x time, y: batch x channel x time OR batch x time 

58 if x.shape[-1] != y.shape[-1]: 

59 raise ValueError( 

60 "Time dimensions of x and y tensors must " 

61 f"be the same, found {x.shape[-1]} and {y.shape[-1]}" 

62 ) 

63 elif x.ndim == 1 and not y.ndim == 1: 

64 raise ValueError( 

65 "Can't compute cross spectral density of " 

66 f"1D tensor x with {y.ndim}D tensor y" 

67 ) 

68 elif x.ndim > 1 and y.ndim == x.ndim: 

69 if not y.shape == x.shape: 

70 raise ValueError( 

71 "If x and y tensors have the same number " 

72 "of dimensions, shapes must fully match. " 

73 f"Found shapes {x.shape} and {y.shape}" 

74 ) 

75 elif x.ndim > 1 and y.ndim != (x.ndim - 1): 

76 raise ValueError( 

77 "Can't compute cross spectral density of " 

78 f"tensors with shapes {x.shape} and {y.shape}" 

79 ) 

80 elif x.ndim > 2 and y.shape[0] != x.shape[0]: 

81 raise ValueError( 

82 "If x is a 3D tensor and y is a 2D tensor, " 

83 "0th batch dimensions must match, but found " 

84 f"values {x.shape[0]} and {y.shape[0]}" 

85 )

88def fast_spectral_density( 

89 x: TimeSeries1to3d, 

90 nperseg: int, 

91 nstride: int, 

92 window: Float[Tensor, " {nperseg//2+1}"], 

93 scale: float, 

94 average: str = "median", 

95 y: TimeSeries1to3d | None = None, 

96) -> FrequencySeries1to3d: 

97 """ 

98 Compute the power spectral density of a multichannel 

99 timeseries or a batch of multichannel timeseries, or 

100 the cross power spectral density of two such timeseries. 

101 This implementation is non-exact for the two lowest 

102 frequency bins, since it implements centering using the 

103 mean for the entire timeseries rather than on a per- 

104 window basis. The benefit of this is a faster implementation, 

105 which might be beneficial for cases where the lowest 

106 frequency components are going to be discarded anyway. 

107 

108 Args: 

109 x: 

110 The timeseries tensor whose power spectral density 

111 to compute, or for cross spectral density the 

112 timeseries whose fft will be conjugated. Can have 

113 shape ``(batch_size, num_channels, length * sample_rate)``, 

114 ``(num_channels, length * sample_rate)``, or 

115 ``(length * sample_rate)``. 

116 nperseg: 

117 Number of samples included in each FFT window 

118 nstride: 

119 Stride between FFT windows 

120 window: 

121 Window array to multiply by each FFT window before 

122 FFT computation. Should have length ``nperseg // 2 + 1``. 

123 scale: 

124 Scale factor to multiply the FFT'd data by, related to 

125 desired units for output tensor (e.g. letting this equal 

126 ``1 / (sample_rate * (window**2).sum())`` will give output 

127 units of density, :math``\\text{Hz}^-1``. 

128 average: 

129 How to aggregate the contributions of each FFT window to 

130 the spectral density. Allowed options are ``'mean'`` and 

131 ``'median'``. 

132 y: 

133 Timeseries tensor to compute cross spectral density 

134 with ``x``. If left as ``None``, ``x``'s power spectral 

135 density will be returned. Otherwise, if ``x`` is 1D, 

136 ``y`` must also be 1D. If ``x`` is 2D, the assumption 

137 is that this represents a single multi-channel timeseries, 

138 and ``y`` must be either 2D or 1D. In the former case, 

139 the cross-spectral densities of each channel will be 

140 computed individually, so ``y`` must have the same shape as ``x``. 

141 Otherwise, this will compute the CSD of each of ``x``'s channels 

142 with ``y``. If ``x`` is 3D, this will be assumed to be a batch 

143 of multi-channel timeseries. In this case, ``y`` can either 

144 be 3D, in which case each channel of each batch element will 

145 have its CSD calculated or 2D, which has two different options. 

146 If ``y``'s 0th dimension matches ``x``'s 0th dimension, it will 

147 be assumed that ``y`` represents a batch of 1D timeseries, and 

148 for each batch element this timeseries will have its CSD with 

149 each channel of the corresponding batch element of ``x`` 

150 calculated. Otherwise, it sill be assumed that ``y`` represents 

151 a single multi-channel timeseries, in which case each channel 

152 of ``y`` will have its CSD calculated with the corresponding 

153 channel in ``x`` across _all_ of ``x``'s batch elements. 

154 Returns: 

155 Tensor of power spectral densities of ``x`` or its cross spectral 

156 density with the timeseries in ``y``. 

157 """ 

158 

159 _validate_shapes(x, nperseg, y) 

160 

161 if x.ndim > 2: 

162 # stft only works on 2D input, so roll the 

163 # channel dimension out along the batch 

164 batch_size = x.shape[0] 

165 num_channels = x.shape[1] 

166 x = x.reshape(-1, x.shape[-1]) 

167 

168 if y is not None and y.ndim == 3: 

169 # do the same thing for y if necessary 

170 y = y.reshape(-1, x.shape[-1]) 

171 else: 

172 # this is just a multichannel timeseries, so 

173 # we can ignore batch reshapes later 

174 batch_size = None 

175 

176 x = x - x.mean(axis=-1, keepdims=True) 

177 fft = torch.stft( 

178 x, 

179 n_fft=nperseg, 

180 hop_length=nstride, 

181 window=window, 

182 normalized=False, 

183 center=False, 

184 return_complex=True, 

185 ) 

186 

187 other = fft 

188 if y is not None: 

189 y = y - y.mean(axis=-1, keepdims=True) 

190 y_fft = torch.stft( 

191 y, 

192 n_fft=nperseg, 

193 hop_length=nstride, 

194 window=window, 

195 normalized=False, 

196 center=False, 

197 return_complex=True, 

198 ) 

199 if batch_size is not None and fft.shape[0] != y_fft.shape[0]: 

200 # x is batched but y is not batched, so expand x's 0th 

201 # dimension back out so that we can multiply these arrays 

202 nfreq = nperseg // 2 + 1 

203 fft = fft.reshape(batch_size, num_channels, nfreq, -1) 

204 y_fft = y_fft[:, None] 

205 other = y_fft 

206 

207 fft = torch.conj(fft) * other 

208 if y is None: 

209 fft = fft.real 

210 

211 # some overly complex logic to find the frequency dimension 

212 # to apply this inscrutable signal processing step of 

213 # multiplying the non-edge frequency bins by 2 

214 # TODO: move this after aggregation since it won't affect 

215 # output value, and we have one less axis to do it over? 

216 stop = None if nperseg % 2 else -1 

217 if x.ndim == 1: 

218 fft[1:stop] *= 2 

219 elif fft.ndim < 4: 

220 fft[:, 1:stop] *= 2 

221 else: 

222 fft[:, :, 1:stop] *= 2 

223 fft *= scale 

224 

225 if average == "mean": 

226 fft = fft.mean(axis=-1) 

227 else: 

228 # if this is a cross spectral density, fft will 

229 # be complex and so we'll compute the median along 

230 # both axes 

231 if y is not None: 

232 real_median = median(fft.real, -1) 

233 imag_median = 1j * median(fft.imag, -1) 

234 fft = real_median + imag_median 

235 else: 

236 fft = median(fft, -1) 

237 

238 if fft.ndim == 2 and batch_size is not None: 

239 # if we still haven't expanded the batch dimension 

240 # back out, do so now 

241 fft = fft.reshape(batch_size, num_channels, -1) 

242 return fft

245def spectral_density( 

246 x: TimeSeries1to3d, 

247 nperseg: int, 

248 nstride: int, 

249 window: Float[Tensor, " {nperseg//2+1}"], 

250 scale: float, 

251 average: str = "median", 

252) -> FrequencySeries1to3d: 

253 """ 

254 Compute the power spectral density of a multichannel 

255 timeseries or a batch of multichannel timeseries. 

256 This implementation is exact for all frequency bins, but 

257 slower than the fast implementation. 

258 

259 Args: 

260 x: 

261 The timeseries tensor whose power spectral density 

262 to compute, or for cross spectral density the 

263 timeseries whose fft will be conjugated. Can have 

264 shape ``(batch_size, num_channels, length * sample_rate)``, 

265 ``(num_channels, length * sample_rate)``, or 

266 ``(length * sample_rate)``. 

267 nperseg: 

268 Number of samples included in each FFT window 

269 nstride: 

270 Stride between FFT windows 

271 window: 

272 Window array to multiply by each FFT window before 

273 FFT computation. Should have length ``nperseg // 2 + 1``. 

274 scale: 

275 Scale factor to multiply the FFT'd data by, related to 

276 desired units for output tensor (e.g. letting this equal 

277 ``1 / (sample_rate * (window**2).sum())`` will give output 

278 units of density, :math:`\\text{Hz}^-1`. 

279 average: 

280 How to aggregate the contributions of each FFT window to 

281 the spectral density. Allowed options are ``'mean'`` and 

282 ``'median'``. 

283 """ 

284 

285 _validate_shapes(x, nperseg) 

286 

287 # for non-fast implementation, we need to unfold 

288 # the tensor along the time dimension ourselves 

289 # to detrend each segment individually, so start 

290 # by converting x to a 4D tensor so we can use 

291 # torch's Unfold op 

292 if x.ndim == 1: 

293 reshape = [] 

294 x = x[None, None, None, :] 

295 elif x.ndim == 2: 

296 reshape = [len(x)] 

297 x = x[None, :, None, :] 

298 elif x.ndim == 3: 

299 reshape = list(x.shape[:-1]) 

300 x = x[:, :, None, :] 

301 

302 # calculate the number of segments and trim x along 

303 # the time dimensions so that we can unfold it exactly 

304 num_segments = (x.shape[-1] - nperseg) // nstride + 1 

305 stop = (num_segments - 1) * nstride + nperseg 

306 x = x[..., :stop] 

307 

308 # unfold x into overlapping segments and detrend and window 

309 # each one individually before computing the rfft. Unfold 

310 # will produce a batch x (num_channels * num_segments) x nperseg 

311 # shaped tensor 

312 unfold_op = torch.nn.Unfold((1, num_segments), dilation=(1, nstride)) 

313 x = unfold_op(x) 

314 x = x - x.mean(axis=-1, keepdims=True) 

315 x *= window 

316 

317 # after the fft, we'll have a 

318 # batch x (num_channels * num_segments) x nfreq 

319 # sized tensor 

320 fft = torch.fft.rfft(x, axis=-1).abs() ** 2 

321 

322 if nperseg % 2: 

323 fft[..., 1:] *= 2 

324 else: 

325 fft[..., 1:-1] *= 2 

326 fft *= scale 

327 

328 # unfold the batch and channel dimensions back 

329 # out if there were any to begin with, putting 

330 # the segment dimension as the second to last 

331 reshape += [num_segments, -1] 

332 fft = fft.reshape(*reshape) 

333 

334 if average == "mean": 

335 return fft.mean(axis=-2) 

336 else: 

337 return median(fft, -2)

340def truncate_inverse_power_spectrum( 

341 psd: PSDTensor, 

342 fduration: Float[Tensor, " time"] | float, 

343 sample_rate: float, 

344 highpass: float | None = None, 

345 lowpass: float | None = None, 

346) -> PSDTensor: 

347 """ 

348 Truncate the length of the time domain response 

349 of a whitening filter built using the specified 

350 ``psd`` so that it has maximum length ``fduration`` 

351 seconds. This is meant to mitigate the impact 

352 of sharp features in the background PSD causing 

353 time domain responses longer than the segments 

354 to which the whitening filter will be applied. 

355 

356 Implementation details adapted from 

357 `here <https://github.com/vivinousi/gw-detection-deep-learning/blob/203966cc2ee47c32c292be000fb009a16824b7d9/modules/whiten.py#L8>`_. 

358 

359 Args: 

360 psd: 

361 The one-sided power spectral density used 

362 to construct a whitening filter. 

363 fduration: 

364 Desired length in seconds of the time domain 

365 response of a whitening filter built using 

366 this PSD, or a window of this length to taper 

367 the edges of the time domain response of the 

368 filter. If passed as a float, a Hann window 

369 of this length will be used. 

370 sample_rate: 

371 Rate at which the time domain data to which 

372 the whitening filter will be applied has been 

373 sampled. 

374 highpass: 

375 If specified, will zero out the frequency response 

376 of all frequencies below this value in Hz. If left 

377 as ``None``, no highpass filtering will be applied. 

378 lowpass: 

379 If specified, will zero out the frequency response 

380 of all frequencies above this value in Hz. If left 

381 as ``None``, no lowpass filtering will be applied. 

382 Returns: 

383 The PSD with its time domain response truncated 

384 to ``fduration`` and any filtered frequencies 

385 tapered. 

386 """ # noqa: E501 

387 

388 num_freqs = psd.size(-1) 

389 N = (num_freqs - 1) * 2 

390 

391 # use the inverse of the ASD as the 

392 # impulse response function 

393 inv_asd = 1 / psd**0.5 

394 

395 # zero out frequencies if we want the filter 

396 # to perform highpass/lowpass filtering 

397 df = sample_rate / N 

398 if highpass is not None: 

399 idx = int(highpass / df) 

400 inv_asd[:, :, :idx] = 0 

401 if lowpass is not None: 

402 idx = int(lowpass / df) 

403 inv_asd[:, :, idx:] = 0 

404 

405 if inv_asd.size(-1) % 2: 

406 inv_asd[:, :, -1] = 0 

407 

408 # now convert to time domain representation 

409 q = torch.fft.irfft(inv_asd, n=N, norm="forward", dim=-1) 

410 

411 # taper the edges of the TD filter 

412 if isinstance(fduration, Tensor): 

413 pad = fduration.size(-1) // 2 

414 window = fduration 

415 else: 

416 pad = int(fduration * sample_rate / 2) 

417 window = torch.hann_window(2 * pad, dtype=torch.float64) 

418 window = window.to(q.device) 

419 

420 # 0 out anything else between the tapering regions 

421 q[:, :, :pad] *= window[-pad:] 

422 q[:, :, -pad:] *= window[:pad] 

423 if 2 * pad < q.size(-1): 

424 q[:, :, pad : q.size(-1) - pad] = 0 

425 

426 # convert back to the frequency domain 

427 # to build the desired PSD 

428 inv_asd = torch.fft.rfft(q, n=N, norm="forward", dim=-1) 

429 inv_psd = inv_asd * inv_asd.conj() 

430 psd = 1 / inv_psd.abs() 

431 return psd / 2

434def normalize_by_psd( 

435 X: WaveformTensor, 

436 psd: PSDTensor, 

437 sample_rate: float, 

438 pad: int, 

439 crop: bool = True, 

440): 

441 # compute the FFT of the section we want to whiten 

442 # and divide it by the ASD of the background section. 

443 # If the ASD of any background bin hit inf, set the 

444 # corresponding bin to 0 

445 X = X - X.mean(-1, keepdims=True) 

446 X_tilde = torch.fft.rfft(X.double(), norm="forward", dim=-1) 

447 X_tilde = X_tilde / psd**0.5 

448 X_tilde[torch.isnan(X_tilde)] = 0 

449 

450 # convert back to the time domain and normalize 

451 # TODO: what's this normalization factor? 

452 X = torch.fft.irfft(X_tilde, n=X.shape[-1], norm="forward", dim=-1) 

453 X = X.float() / sample_rate**0.5 

454 

455 # slice off corrupted data at edges of kernel 

456 if crop: 

457 X = X[:, :, pad:-pad] 

458 return X

461def whiten( 

462 X: WaveformTensor, 

463 psd: PSDTensor, 

464 fduration: Float[Tensor, " time"] | float, 

465 sample_rate: float, 

466 highpass: float | None = None, 

467 lowpass: float | None = None, 

468 crop: bool = True, 

469) -> WaveformTensor: 

470 """ 

471 Whiten a batch of timeseries using the specified 

472 background one-sided power spectral densities (PSDs), 

473 modified to have the desired time domain response length 

474 ``fduration`` and possibly to highpass/lowpass filter. 

475 

476 Args: 

477 X: 

478 batch of multichannel timeseries to whiten 

479 psd: 

480 PSDs use to whiten the data. The frequency 

481 response of the whitening filter will be roughly 

482 the inverse of the square root of this PSD, ensuring 

483 that data from the same distribution will have 

484 approximately uniform power after whitening. 

485 If 2D, each batch element in ``X`` will be whitened 

486 using the same PSDs. If 3D, each batch element will 

487 be whitened by the PSDs contained along the 0th 

488 dimenion of ``psd``, and so the first two dimensions 

489 of ``X`` and ``psd`` should match. 

490 fduration: 

491 Desired length in seconds of the time domain 

492 response of a whitening filter built using 

493 this PSD, or a window of this length to taper 

494 the edges of the time domain response of the 

495 filter. If passed as a float, a Hann window 

496 of this length will be used. Moreover, half of 

497 this length will be removed from each edge of 

498 the whitened timeseries to account for filter 

499 settle-in time. 

500 sample_rate: 

501 Rate at which the data in ``X`` has been sampled 

502 highpass: 

503 The frequency in Hz at which to highpass filter 

504 the data, setting the frequency response in the 

505 whitening filter to 0. If left as ``None``, no 

506 highpass filtering will be applied. 

507 lowpass: 

508 The frequency in Hz at which to lowpass filter 

509 the data, setting the frequency response in the 

510 whitening filter to 0. If left as ``None``, no 

511 lowpass filtering will be applied. 

512 crop: 

513 If ``True``, crop ``fduration / 2`` seconds of data 

514 from both sides of the time dimension to remove the 

515 corruption from the filter. If ``False``, return the 

516 full timeseries. 

517 Returns: 

518 Batch of whitened multichannel timeseries with ``fduration / 2`` 

519 seconds optionally trimmed from each side. 

520 """ 

521 

522 # figure out how much data we'll need to slice 

523 # off after whitening 

524 if isinstance(fduration, Tensor): 

525 pad = fduration.size(-1) // 2 

526 else: 

527 pad = int(fduration * sample_rate / 2) 

528 

529 N = X.size(-1) 

530 if N <= (2 * pad): 

531 raise ValueError( 

532 f"Not enough timeseries samples {N} for number of " 

533 f"padded samples {2 * pad}" 

534 ) 

535 

536 # normalize the number of expected dimensions in the PSD 

537 while psd.ndim < 3: 

538 psd = psd[None] 

539 

540 # possibly interpolate our PSD to match the number 

541 # of frequency bins we expect to get from X 

542 num_freqs = N // 2 + 1 

543 if psd.size(-1) != num_freqs: 

544 # TODO: does there need to be any rescaling to 

545 # keep the integral of the PSD constant? 

546 psd = torch.nn.functional.interpolate( 

547 psd, size=(num_freqs,), mode="linear" 

548 ) 

549 

550 # truncate it to have the desired 

551 # time domain response length 

552 psd = truncate_inverse_power_spectrum( 

553 psd, 

554 fduration, 

555 sample_rate, 

556 highpass, 

557 lowpass, 

558 ) 

559 

560 return normalize_by_psd(X, psd, sample_rate, pad, crop)
