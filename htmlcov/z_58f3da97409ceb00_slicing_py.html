13def unfold_windows( 

14 x: TimeSeries1to3d, 

15 window_size: int, 

16 stride: int, 

17 drop_last: bool = True, 

18) -> Union[ 

19 Float[TimeSeries1d, " window"], 

20 Float[TimeSeries2d, " window"], 

21 Float[TimeSeries3d, " window"], 

22]: 

23 """Unfold a timeseries into windows 

24 

25 Args: 

26 x: 

27 The timeseries to unfold. Can have shape 

28 `(batch_size, num_channels, length * sample_rate)`, 

29 `(num_channels, length * sample_rate)`, or 

30 `(length * sample_rate)` 

31 window_size: 

32 The size of the windows to unfold from x 

33 stride: 

34 The stride between windows 

35 drop_last: 

36 If true, does not return the remainder that exists 

37 when the timeseries cannot be evenly broken up into 

38 windows 

39 

40 Returns: 

41 A tensor of shape 

42 `(num_windows, batch_size, num_channels, kernel_size)`, 

43 `(num_windows, num_channels, kernel_size)`, or 

44 `(num_windows, kernel_size)` depending on whether the 

45 input tensor is 3D, 2D, or 1D 

46 

47 If `drop_last` is false, returns the remainder of the 

48 timeseries, shaped to be compatible with the returned 

49 unfolded tensor 

50 """ 

51 

52 num_windows = (x.shape[-1] - window_size) // stride + 1 

53 remainder = x.shape[-1] - window_size - (num_windows - 1) * stride 

54 if remainder == 0: 

55 if not drop_last: 

56 # create an empty tensor for consistency 

57 shape = list(x.shape) 

58 shape[-1] = 0 

59 remainder = torch.zeros(shape, dtype=x.dtype, device=x.device) 

60 else: 

61 # separate x from its remainder _regardless_ of whether 

62 # we end up returning it or not 

63 x, remainder = torch.split( 

64 x, [x.shape[-1] - remainder, remainder], dim=-1 

65 ) 

66 

67 reshape = list(x.shape[:-1]) 

68 if x.ndim == 1: 

69 x = x[None, None, None, :] 

70 elif x.ndim == 2: 

71 x = x[None, :, None, :] 

72 elif x.ndim == 3: 

73 x = x[:, :, None, :] 

74 

75 x = unfold(x, (1, num_windows), dilation=(1, stride)) 

76 reshape += [num_windows, -1] 

77 x = x.reshape(*reshape) 

78 x = x.transpose(1, -2).transpose(0, 1) 

79 

80 if not drop_last: 

81 return x, remainder[None] 

82 return x

85def slice_kernels( 

86 x: TimeSeries1to3d, 

87 idx: Int64[Tensor, "..."], 

88 kernel_size: int, 

89) -> BatchTimeSeriesTensor: 

90 """Slice kernels from single or multichannel timeseries 

91 

92 Given a 1D timeseries or a 2D tensor representing a 

93 multichannel timeseries, slice kernels of a given size 

94 from the timeseries starting at the indicated indices. 

95 Returns a batch of 1D or 2D kernels, and so will have 

96 one more dimension than `x`. 

97 

98 Args: 

99 x: 

100 The timeseries tensor to slice kernels from 

101 idx: 

102 The indices in `x` of the first sample of each 

103 kernel. If `x` is 1D, `idx` must be 1D as well. 

104 If `x` is 2D and `idx` is 1D, `idx` is assumed 

105 to represent the first index of the kernels sliced 

106 from _all_ channels (i.e. the channels are sliced 

107 coincidentally). If `x` is 2D and `idx` is also 2D, 

108 `idx` should have shape `(batch_size, num_channels)`, 

109 and its values are assumed to represent the first index 

110 of the kernel sliced from each channel _independently_. 

111 If `x` is 3D, `idx` _must_ be 1D, and have the same length 

112 as `x`. In this case, it is assumed that the elements of 

113 `idx` represent the starting index in the last dimension 

114 of `x` from which to sample a batch of kernels 

115 coincidentally among the channels. 

116 kernel_size: 

117 The length of the kernels to slice from the timeseries 

118 

119 Returns: 

120 A tensor of shape `(batch_size, kernel_size)` if `x` is 

121 1D and `(batch_size, num_channels, kernel_size)` if `x` 

122 is 2D, where `batch_size = idx.shape[0]` and 

123 `num_channels = x.shape[0]` if `x` is 2D. 

124 """ 

125 

126 # create the indices all the slices will be built around, 

127 # and ensure they live on the appropriate device 

128 kernels = torch.arange(kernel_size, device=x.device) 

129 idx = idx.to(kernels.device) 

130 

131 # TODO: add try-catches aroud the actual slicing operations 

132 # to catch out-of-range index errors and raise with a 

133 # standardized error that's very explicit 

134 if x.ndim == 1: 

135 if idx.ndim != 1: 

136 raise ValueError( 

137 f"idx tensor has {idx.ndim} dimensions, expected 1" 

138 ) 

139 

140 # this is a one dimensional array that we want 

141 # to select kernels from beginning at each of 

142 # the specified indices 

143 kernels = kernels.view(kernel_size, 1) 

144 kernels = kernels.repeat(1, len(idx)) 

145 kernels = (kernels + idx).t() 

146 return torch.take(x, kernels) 

147 elif x.ndim == 2 and idx.ndim == 1: 

148 # this is a multi-channel timeseries and we want 

149 # to select a set of kernels from the channels 

150 # coincidentally 

151 

152 # channels x batch_size x kernel_size 

153 kernels = kernels.view(1, 1, kernel_size) 

154 kernels = kernels.repeat(len(x), len(idx), 1) 

155 kernels += idx.view(1, -1, 1) 

156 

157 # channels x (batch_size * kernel_size) 

158 kernels = kernels.reshape(len(x), -1) 

159 

160 # channels x (batch_size * kernel_size) 

161 # (only this time it's data from x) 

162 x = torch.take_along_dim(x, kernels, axis=1) 

163 

164 # channels x batch_size x kernel_size 

165 x = x.reshape(len(x), len(idx), kernel_size) 

166 

167 # batch_size x channels x kernel_size 

168 return x.transpose(0, 1) 

169 elif x.ndim == 2 and idx.ndim == 2: 

170 # this is a multi-channel timeseries and we want 

171 # to select _different_ kernels from each channel 

172 if len(x) != idx.shape[1]: 

173 raise ValueError( 

174 ( 

175 "Can't slice array with shape {} with indices " 

176 "with shape {}" 

177 ).format(x.shape, idx.shape) 

178 ) 

179 

180 # batch_size x num_channels x kernel_size 

181 kernels = kernels.view(1, 1, kernel_size) 

182 kernels = kernels.repeat(len(idx), len(x), 1) 

183 kernels += idx[:, :, None] 

184 

185 # num_channels x (batch_size * kernel_size) 

186 kernels = kernels.transpose(0, 1) 

187 kernels = kernels.reshape(len(x), -1) 

188 

189 # num_channels x (batch_size * kernel_size) 

190 # sample a batch's worth of kernels for each channel 

191 x = torch.take_along_dim(x, kernels, axis=1) 

192 

193 # num_channels x batch_size x kernel_size 

194 x = x.reshape(len(x), len(idx), kernel_size) 

195 

196 # batch_size x num_channels x kernel_size 

197 return x.transpose(0, 1) 

198 elif x.ndim == 2: 

199 raise ValueError( 

200 f"Can't slice 2D array with indices with {idx.ndim} dimensions" 

201 ) 

202 elif x.ndim == 3 and idx.ndim == 1: 

203 # slice a single kernel coincidentally from a batch 

204 # of multichannel timeseries 

205 if len(idx) != len(x): 

206 raise ValueError( 

207 "Can't slice kernels from batch of length {} " 

208 "using indices of length {}".format(len(x), len(idx)) 

209 ) 

210 

211 # batch_size x kernel_size 

212 kernels = kernels.view(1, -1) 

213 kernels = kernels.repeat(len(idx), 1) 

214 kernels += idx.view(-1, 1) 

215 

216 # batch_size x num_channels x kernel_size 

217 kernels = kernels.view(-1, 1, kernel_size) 

218 kernels = kernels.repeat(1, x.shape[1], 1) 

219 

220 return torch.take_along_dim(x, kernels, axis=-1) 

221 elif x.ndim == 3: 

222 raise ValueError( 

223 f"Can't slice 3D array with indices with {idx.ndim} dimensions" 

224 ) 

225 else: 

226 raise ValueError( 

227 f"Can't slice kernels from tensor with shape {x.shape}" 

228 )

231def sample_kernels( 

232 X: TimeSeries1to3d, 

233 kernel_size: int, 

234 N: Optional[int] = None, 

235 max_center_offset: Optional[int] = None, 

236 coincident: bool = True, 

237) -> BatchTimeSeriesTensor: 

238 """Randomly sample kernels from a single or multichannel timeseries 

239 

240 For a tensor representing one or multiple channels of 

241 timeseries data, randomly slice kernels of a fixed 

242 length from the timeseries. If `X` is 1D, kernels will 

243 be sampled uniformly from `X`. If `X` is 2D, kernels 

244 will be sampled from the first dimension of `X` (assumed 

245 to be the time dimension) in a manner that depends on the 

246 values of the `max_center_offset` and `coincident` kwargs. 

247 If `X` is 3D, one kernel will be sampled coincidentally from 

248 each element along the 0th axis of `X`. In this case, `N` must 

249 either be `None` or be equal to `len(X)`. 

250 

251 Args: 

252 X: 

253 The timeseries tensor from which to sample kernels 

254 kernel_size: The size of the kernels to sample 

255 N: 

256 The number of kernels to sample. Can be left as 

257 `None` if `X` is 3D, otherwise must be specified 

258 max_center_offeset: 

259 If `X` is 2D, this indicates the maximum distance 

260 from the center of the timeseries the edge of 

261 sampled kernels may fall. If left as `None`, kernels 

262 will be sampled uniformly across all of `X`'s time 

263 dimension. If greater than 0, defines the maximum 

264 distance that the rightmost edge of the kernel may 

265 fall from the center of the timeseries (the leftmost 

266 edge will always be sampled such that the center of 

267 the timeseries falls within or afer the kernel). If 

268 equal to 0, every kernel sampled will contain the center 

269 of the timeseries, which may fall anywhere within the 

270 kernel with uniform probability. If less than 0, defines 

271 the minimum distance that the center of the timeseries 

272 must fall from either edge of the kernel. If `X` is 

273 1D, this argument is ignored. 

274 coincident: 

275 If `X` is 2D, determines whether the individual channels 

276 of `X` sample the same kernels or different kernels 

277 independently, i.e. whether the channels of each batch 

278 element in the output will contain coincident data. If 

279 `X` is 1D, this argument is ignored. 

280 Returns: 

281 A batch of sampled kernels. If `X` is 1D, this will have 

282 shape `(N, kernel_size)`. If `X` is 2D, this will have 

283 shape `(N, num_channels, kernel_size)`, where 

284 `num_channels = X.shape[0]`. 

285 """ 

286 

287 if X.shape[-1] < kernel_size: 

288 raise ValueError( 

289 "Can't sample kernels of size {} from tensor with shape {}".format( 

290 kernel_size, X.shape 

291 ) 

292 ) 

293 elif X.ndim > 3: 

294 raise ValueError( 

295 f"Can't sample kernels from tensor with {X.ndim} dimensions" 

296 ) 

297 elif X.ndim < 3 and N is None: 

298 raise ValueError( 

299 "Must specify number of kernels N if X has fewer than 3 dimensions" 

300 ) 

301 elif X.ndim == 3 and N is not None and N != len(X): 

302 raise ValueError( 

303 ( 

304 "Can't sample {} kernels from 3D tensor with " 

305 "batch dimension {}" 

306 ).format(N, len(X)) 

307 ) 

308 

309 if X.ndim == 1: 

310 idx = torch.randint(len(X) - kernel_size, size=(N,)) 

311 return slice_kernels(X, idx, kernel_size) 

312 

313 center = int(X.shape[-1] // 2) 

314 

315 if max_center_offset is None: 

316 # sample uniformly from all of X's time dimension 

317 min_val, max_val = 0, X.shape[-1] - kernel_size 

318 elif max_center_offset >= 0: 

319 # a positive max_center_offset means we're allowed 

320 # to put some space between the center of the timeseries 

321 # and the right edge of the kernel, but not the center 

322 # and the left edge of the kernel 

323 min_val = center - max_center_offset - kernel_size 

324 max_val = center 

325 else: 

326 # a negative max_center_offset means that we want 

327 # to enforce that the center of the timeseries is 

328 # some distance away from the edge of the kernel 

329 min_val = center - max_center_offset - kernel_size 

330 max_val = center + max_center_offset 

331 

332 if max_val <= min_val: 

333 # if our required offset value is more than half 

334 # the kernel length, we won't be able to sample 

335 # any kernels at all 

336 raise ValueError( 

337 "Negative center offset value {} is too large " 

338 "for requested kernel size {}".format( 

339 max_center_offset, kernel_size 

340 ) 

341 ) 

342 

343 if min_val < 0: 

344 # if kernel_size > center - max_center_offset, 

345 # we may end up with negative indices 

346 raise ValueError( 

347 ( 

348 "Kernel size {} is too large for requested center " 

349 "offset value {}" 

350 ).format(kernel_size, max_center_offset) 

351 ) 

352 

353 if X.ndim == 3 or coincident: 

354 # sampling coincidentally, so just need a single 

355 # index for each element in the output batch 

356 N = N or len(X) 

357 shape = (N,) 

358 else: 

359 # otherwise, each channel in each batch sample 

360 # will require its own sampling index 

361 shape = (N, len(X)) 

362 

363 idx = torch.randint(min_val, max_val, size=shape).to(X.device) 

364 return slice_kernels(X, idx, kernel_size)
