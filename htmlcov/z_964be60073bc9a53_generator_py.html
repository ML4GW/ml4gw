20class TimeDomainCBCWaveformGenerator(torch.nn.Module): 

21 """ 

22 Waveform generator that generates time-domain waveforms from 

23 frequency-domain approximants. 

24 

25 Frequency domain waveforms are conditioned as done by lalsimulation. 

26 Specifically, waveforms are generated with a starting frequency `fstart` 

27 slightly below the requested `f_min`, so that they can be tapered from 

28 `fstart` to `f_min` using a cosine window. 

29 

30 Please see https://lscsoft.docs.ligo.org/lalsuite/lalsimulation/group___l_a_l_sim_inspiral__c.html#gac9f16dab2cbca5a431738ee7d2505969 

31 for more information 

32 

33 Args: 

34 approximant: 

35 A callable that returns hplus and hcross polarizations 

36 given requested frequencies and relevant set of parameters. 

37 See `ml4gw.waveforms.cbc` for implemented approximants. 

38 sample_rate: 

39 Rate at which returned time domain waveform will be 

40 sampled in Hz. This also specifies `f_max` for generating 

41 waveforms via the nyquist frequency: `f_max = sample_rate // 2`. 

42 f_min: 

43 Lower frequency bound for waveforms 

44 duration: 

45 Length of waveform in seconds. 

46 Waveforms will be left padded with zeros 

47 appropiately to fill the requested duration 

48 right_pad: 

49 How far from the right edge of the window 

50 in seconds the returned waveform coalescence 

51 will be placed. 

52 f_ref: 

53 Reference frequency for the waveform 

54 """ # noqa: E501

92 def build_highpass_filter(self): 

93 """ 

94 Builds highpass filter object. 

95 

96 See https://git.ligo.org/lscsoft/lalsuite/-/blob/master/lalsimulation/python/lalsimulation/gwsignal/core/conditioning_subroutines.py?ref_type=heads#L10 

97 """ # noqa: E501 

98 order = 8.0 

99 w1 = np.tan(np.pi * (self.f_min) / self.sample_rate) 

100 attenuation = 0.99 

101 wc = w1 * (1.0 / attenuation**0.5 - 1) ** (1.0 / (2.0 * order)) 

102 fc = self.sample_rate * np.arctan(wc) / np.pi 

103 

104 return IIRFilter( 

105 order, 

106 fc / (self.sample_rate / 2), 

107 btype="highpass", 

108 ftype="butterworth", 

109 )

116 def generate_conditioned_fd_waveform( 

117 self, **parameters: dict[str, BatchTensor] 

118 ) -> Tuple[Float[Tensor, "{N} samples"], Float[Tensor, "{N} samples"]]: 

119 """ 

120 Generate a conditioned frequency domain waveform from a 

121 frequency-domain approximant. 

122 

123 Based on https://git.ligo.org/lscsoft/lalsuite/-/blob/master/lalsimulation/python/lalsimulation/gwsignal/core/waveform_conditioning.py?ref_type=heads#L248 

124 

125 Args: 

126 **parameters: 

127 Dictionary of parameters for waveform generation where key is 

128 the parameter name and value is a tensor of parameters. 

129 It is required that `parameters` contains `mass_1`, `mass_2`, 

130 `s1z`, and `s2z` keys, which are used for determining 

131 parameters of data conditioning. 

132 

133 If the specified approximant takes other parameters for 

134 waveform generation, like `chirp_mass` and `mass_ratio`, the 

135 utility functions in `ml4gw.waveforms.conversion`may be useful 

136 for populating the parameters dictionary with these 

137 additional parameters. 

138 

139 Note that, if using an approximant from `ml4gw.waveforms.cbc`, 

140 any additional keys in `parameters` not ingested by the 

141 approximant will be ignored. 

142 """ # noqa: E501 

143 # convert masses to kg, make sure 

144 # they are doubles so there is no 

145 # overflow in the calculations 

146 mass_1, mass_2 = ( 

147 parameters["mass_1"].double() * MSUN, 

148 parameters["mass_2"].double() * MSUN, 

149 ) 

150 total_mass = mass_1 + mass_2 

151 s1z, s2z = parameters["s1z"], parameters["s2z"] 

152 device = mass_1.device 

153 

154 f_isco = utils.frequency_isco(mass_1, mass_2) 

155 f_min = torch.minimum( 

156 f_isco, 

157 torch.tensor(self.f_min, device=device), 

158 ) 

159 

160 # upper bound on chirp time 

161 tchirp = utils.chirp_time_bound(f_min, mass_1, mass_2, s1z, s2z) 

162 

163 # upper bound on final black hole spin 

164 s = utils.final_black_hole_spin_bound(s1z, s2z) 

165 

166 # upper bound on the final plunge, merger, and ringdown time 

167 tmerge = utils.merge_time_bound( 

168 mass_1, mass_2 

169 ) + utils.ringdown_time_bound(total_mass, s) 

170 

171 # extra time to include for all waveforms to take care of situations 

172 # where the frequency is close to merger (and is sweeping rapidly): 

173 # this is a few cycles at the low frequency 

174 textra = EXTRA_CYCLES / f_min 

175 

176 # lower bound on chirpt frequency start used for 

177 # conditioning the frequency domain waveform 

178 fstart = utils.chirp_start_frequency_bound( 

179 (1.0 + EXTRA_TIME_FRACTION) * tchirp, mass_1, mass_2 

180 ) 

181 

182 # revised chirp time estimate based on fstart 

183 tchirp_fstart = utils.chirp_time_bound( 

184 fstart, mass_1, mass_2, s1z, s2z 

185 ) 

186 

187 # chirp length in samples 

188 chirplen = torch.round( 

189 (tchirp_fstart + tmerge + 2.0 * textra) * self.sample_rate 

190 ) 

191 

192 # pad to next power of 2 

193 chirplen = 2 ** torch.ceil(torch.log(chirplen) / math.log(2)) 

194 

195 # get smallest df corresponding to longest chirp length, 

196 # which will make sure there is no wrap around effects. 

197 df = 1.0 / (chirplen.max() / self.sample_rate) 

198 

199 # generate frequency array from 0 to nyquist based on df 

200 frequencies = self.get_frequencies(df).to(mass_1.device) 

201 

202 # downselect to frequencies above fstart, 

203 # and generate the waveform at the specified frequencies 

204 freq_mask = frequencies >= fstart.min() 

205 waveform_frequencies = frequencies[freq_mask] 

206 

207 # generate the waveform at specified frequencies 

208 cross, plus = self.approximant( 

209 waveform_frequencies, **parameters, f_ref=self.f_ref 

210 ) 

211 batch_size = cross.size(0) 

212 

213 # create tensors to hold the full spectrum 

214 # of frequencies from 0 to nyquist, and then 

215 # fill in the requested frequencies with the waveform values 

216 shape = (batch_size, frequencies.size(0)) 

217 hc_spectrum = torch.zeros(shape, dtype=cross.dtype, device=device) 

218 hp_spectrum = torch.zeros(shape, dtype=plus.dtype, device=device) 

219 

220 hc_spectrum[:, freq_mask] = cross 

221 hp_spectrum[:, freq_mask] = plus 

222 

223 # build a taper that is dependent on each 

224 # individual waveforms fstart; 

225 # since this means that the taper sizes 

226 # will be different for each waveform, 

227 # construct the tapers based on the maximum size 

228 # and then set the values outside of the individual 

229 # waveform taper regions to 1.0 

230 k0s = torch.round(fstart / df) 

231 k1s = torch.round(f_min / df) 

232 

233 num_freqs = frequencies.size(0) 

234 frequency_indices = torch.arange(num_freqs, device=device) 

235 taper_mask = frequency_indices <= k1s[:, None] 

236 taper_mask &= frequency_indices >= k0s[:, None] 

237 

238 indices = frequency_indices.expand(batch_size, -1) 

239 

240 kvals = indices[taper_mask] 

241 k0s_expanded = k0s.unsqueeze(1).expand(-1, num_freqs)[taper_mask] 

242 k1s_expanded = k1s.unsqueeze(1).expand(-1, num_freqs)[taper_mask] 

243 

244 windows = 0.5 - 0.5 * torch.cos( 

245 torch.pi * (kvals - k0s_expanded) / (k1s_expanded - k0s_expanded) 

246 ) 

247 

248 hc_spectrum[taper_mask] *= windows 

249 hp_spectrum[taper_mask] *= windows 

250 

251 # zero out frequencies below fstart 

252 zero_mask = frequencies < fstart[:, None] 

253 hc_spectrum[zero_mask] = 0 

254 hp_spectrum[zero_mask] = 0 

255 

256 # set nyquist frequency to zero 

257 hc_spectrum[..., -1], hp_spectrum[..., -1] = 0.0, 0.0 

258 

259 # apply time translation in (i.e. phase shift in frequency domain) 

260 # that will translate the coalescense time such that it is `right_pad` 

261 # seconds from the right edge of the window 

262 tshift = round(self.right_pad * self.sample_rate) / self.sample_rate 

263 kvals = torch.arange(num_freqs, device=device) 

264 phase_shift = torch.exp(1j * 2 * torch.pi * df * tshift * kvals) 

265 

266 hc_spectrum *= phase_shift 

267 hp_spectrum *= phase_shift 

268 

269 return hc_spectrum, hp_spectrum

271 def forward( 

272 self, 

273 **parameters, 

274 ) -> Tuple[Float[Tensor, "{N} samples"], Float[Tensor, "{N} samples"]]: 

275 """ 

276 Generates a time-domain waveform from a frequency-domain approximant. 

277 Conditioning is based onhttps://git.ligo.org/lscsoft/lalsuite/-/blob/master/lalsimulation/python/lalsimulation/gwsignal/core/waveform_conditioning.py?ref_type=heads#L248 

278 

279 A frequency domain waveform is generated, conditioned 

280 (see `generate_conditioned_fd_waveform`) and fft'd into the time-domain 

281 

282 **parameters: 

283 Dictionary of parameters for waveform generation where each key is 

284 the parameter name and each value is a tensor of parameters. 

285 It is required that `parameters` contains `mass_1`, `mass_2`, 

286 `s1z`, and `s2z` keys, which are used for determining parameters 

287 of data conditioning. 

288 

289 If the specified approximant takes other parameters for waveform 

290 generation, like `chirp_mass` and `mass_ratio`, the utility 

291 functions in `ml4gw.waveforms.conversion` may be useful for 

292 populating the parameters dictionary with these additional 

293 parameters. 

294 

295 Note that, if using an approximant from `ml4gw.waveforms.cbc`, 

296 any additional keys in `parameters` not ingested by the 

297 approximant will be ignored. 

298 """ # noqa: E501 

299 

300 hc, hp = self.generate_conditioned_fd_waveform(**parameters) 

301 

302 # fft to time domain and apply appropiate scaling 

303 hc = torch.fft.irfft(hc) * self.sample_rate 

304 hp = torch.fft.irfft(hp) * self.sample_rate 

305 

306 # TODO: some additional tapering in the time 

307 # domain is performed in lalsimulation 

308 

309 # pad waveforms on left up to requested duration 

310 pad = int((self.duration * self.sample_rate) - hp.shape[-1]) 

311 hc = torch.nn.functional.pad(hc, (pad, 0)) 

312 hp = torch.nn.functional.pad(hp, (pad, 0)) 

313 

314 # finally, highpass the waveforms, 

315 # going to double precision 

316 hp = self.highpass(hp.double()) 

317 hc = self.highpass(hc.double()) 

318 

319 return hc, hp
