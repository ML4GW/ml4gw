606 def phenom_d_inspiral_phase(

648 def fmaxCalc(self, fRD, fDM, gamma2, gamma3): 

649 mask = gamma2 <= 1 

650 # calculate result for gamma2 <= 1 case 

651 sqrt_term = torch.sqrt(1 - gamma2.pow(2)) 

652 result_case1 = fRD + (fDM * (-1 + sqrt_term) * gamma3) / gamma2 

653 

654 # calculate result for gamma2 > 1 case 

655 # i.e. don't add sqrt term 

656 result_case2 = fRD + (-fDM * gamma3) / gamma2 

657 

658 # combine results using mask 

659 result = torch.where(mask, result_case1, result_case2) 

660 

661 return torch.abs(result)

663 def _linear_interp_finspin(self, finspin): 

664 # chi is a batch of final spins i.e. torch.Size([n]) 

665 right_spin_idx = torch.bucketize(finspin, self.qnmdata_a) 

666 right_spin_val = self.qnmdata_a[right_spin_idx] 

667 # QNMData_a is sorted, hence take the previous index 

668 left_spin_idx = right_spin_idx - 1 

669 left_spin_val = self.qnmdata_a[left_spin_idx] 

670 

671 if not torch.all(left_spin_val < right_spin_val): 

672 raise RuntimeError( 

673 "Left value in grid should be greater than right. " 

674 "Maybe be caused for extremal spin values." 

675 ) 

676 left_fring = self.qnmdata_fring[left_spin_idx] 

677 right_fring = self.qnmdata_fring[right_spin_idx] 

678 slope_fring = right_fring - left_fring 

679 slope_fring /= right_spin_val - left_spin_val 

680 

681 left_fdamp = self.qnmdata_fdamp[left_spin_idx] 

682 right_fdamp = self.qnmdata_fdamp[right_spin_idx] 

683 slope_fdamp = right_fdamp - left_fdamp 

684 slope_fdamp /= right_spin_val - left_spin_val 

685 

686 return ( 

687 slope_fring * (finspin - left_spin_val) + left_fring, 

688 slope_fdamp * (finspin - left_spin_val) + left_fdamp, 

689 )
