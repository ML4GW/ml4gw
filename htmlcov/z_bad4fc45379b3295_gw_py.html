32def outer(x: VectorGeometry, y: VectorGeometry) -> TensorGeometry: 

33 """ 

34 Compute the outer product of two batches of vectors 

35 """ 

36 return torch.einsum("...i,...j->...ij", x, y) 

37 

38 

39def plus(m: VectorGeometry, n: VectorGeometry) -> TensorGeometry: 

40 return outer(m, m) - outer(n, n) 

41 

42 

43def cross(m: VectorGeometry, n: VectorGeometry) -> TensorGeometry: 

44 return outer(m, n) + outer(n, m) 

45 

46 

47def breathing(m: VectorGeometry, n: VectorGeometry) -> TensorGeometry: 

48 return outer(m, m) + outer(n, n)

51polarization_funcs = { 

52 "plus": plus, 

53 "cross": cross, 

54 "breathing": breathing,

58def compute_antenna_responses( 

59 theta: BatchTensor, 

60 psi: BatchTensor, 

61 phi: BatchTensor, 

62 detector_tensors: NetworkDetectorTensors, 

63 modes: List[str], 

64) -> Float[Tensor, "batch polarizations num_ifos"]: 

65 """ 

66 Compute the antenna pattern factors of a batch of 

67 waveforms as a function of the sky parameters of 

68 their sources as well as the detector tensors of 

69 the interferometers whose response is being 

70 calculated. 

71 

72 Args: 

73 theta: 

74 Angle of each source in radians relative 

75 to the celestial equator 

76 psi: 

77 Angle in radians between each source's 

78 natural polarization basis and the basis 

79 which has the 0th unit vector pointing along 

80 the celestial equator 

81 phi: 

82 Angle in radians between each source's right 

83 ascension and the right ascension of the 

84 geocenter 

85 detector_tensors: 

86 Detector tensor for each of the interferometers 

87 for which a response is being calculated, stacked 

88 along the 0th axis 

89 modes: 

90 Which polarization modes to compute the response for 

91 Returns: 

92 A tensor representing interferometer antenna pattern 

93 factors for each of the polarizations of each of 

94 the waveforms, for each interferometer. 

95 """ 

96 

97 # add a dimension so that we can do some tensor 

98 # manipulations batch-wise later 

99 theta = theta.view(-1, 1) 

100 psi = psi.view(-1, 1) 

101 phi = phi.view(-1, 1) 

102 

103 # pre-compute all our trigonometric functions 

104 # since we'll end up using them all at least twice 

105 sin_theta, cos_theta = torch.sin(theta), torch.cos(theta) 

106 sin_psi, cos_psi = torch.sin(psi), torch.cos(psi) 

107 sin_phi, cos_phi = torch.sin(phi), torch.cos(phi) 

108 

109 # u and v are the unit vectors orthogonal to the 

110 # source's propogation vector, where v is 

111 # parallel to the celestial equator 

112 u = torch.cat( 

113 [cos_phi * cos_theta, cos_theta * sin_phi, -sin_theta], axis=1 

114 ) 

115 v = torch.cat([-sin_phi, cos_phi, torch.zeros_like(phi)], axis=1) 

116 

117 # m and n are the unit vectors along the waveforms 

118 # natural polarization basis 

119 m = -u * sin_psi - v * cos_psi 

120 n = -u * cos_psi + v * sin_psi 

121 

122 # compute the polarization tensor of each signal 

123 # as a function of the parameters of its source 

124 polarizations = [] 

125 for mode in modes: 

126 try: 

127 polarization = polarization_funcs[mode](m, n) 

128 except KeyError as exc: 

129 raise ValueError(f"No polarization mode {mode}") from exc 

130 

131 # add a dummy dimension for concatenating 

132 polarizations.append(polarization) 

133 

134 # shape: batch x num_polarizations x 3 x 3 

135 polarization = torch.stack(polarizations, axis=1) 

136 

137 # compute the weight of each interferometer's response 

138 # to each polarization: batch x polarizations x ifos 

139 return torch.einsum("...jk,ijk->...i", polarization, detector_tensors)

142def shift_responses( 

143 responses: WaveformTensor, 

144 theta: BatchTensor, 

145 phi: BatchTensor, 

146 vertices: NetworkVertices, 

147 sample_rate: float, 

148) -> WaveformTensor: 

149 omega = torch.column_stack( 

150 [ 

151 torch.sin(theta) * torch.cos(phi), 

152 torch.sin(theta) * torch.sin(phi), 

153 torch.cos(theta), 

154 ] 

155 ).view(-1, 1, 3) 

156 

157 # compute the time delay between the geocenter 

158 # and each interferometer in units of s, then 

159 # convert to units of samples and discretize. 

160 # Divide by c in the second line so that we only 

161 # need to multiply the array by a single float 

162 dt = -(omega * vertices).sum(axis=-1) 

163 dt *= sample_rate / C 

164 dt = torch.trunc(dt).type(torch.int64) 

165 

166 # rolling by gathering implementation based on 

167 # https://stackoverflow.com/a/68641864 

168 # start by just creating a big arange along the last axis 

169 idx = torch.ones_like(responses).type(torch.int64) 

170 idx = torch.cumsum(idx, axis=-1) - 1 

171 

172 # apply the offset to the indices along the last axis, 

173 # then modulo by the waveform size to make all the 

174 # specified indices legitimate and unique 

175 idx -= dt[:, :, None] 

176 idx %= idx.shape[-1] 

177 

178 # unfortunately I can't figure out how to do this 

179 # last step without doing looping over the ifos 

180 rolled = [] 

181 for i in range(len(vertices)): 

182 ifo = torch.gather(responses[:, i], 1, idx[:, i]) 

183 rolled.append(ifo) 

184 return torch.stack(rolled, axis=1)

187def compute_observed_strain( 

188 dec: BatchTensor, 

189 psi: BatchTensor, 

190 phi: BatchTensor, 

191 detector_tensors: NetworkDetectorTensors, 

192 detector_vertices: NetworkVertices, 

193 sample_rate: float, 

194 **polarizations: Float[Tensor, "batch time"], 

195) -> WaveformTensor: 

196 """ 

197 Compute the strain timeseries $h(t)$ observed by a network 

198 of interferometers from the given polarization timeseries 

199 corresponding to gravitational waveforms from sources with 

200 the indicated sky parameters. 

201 

202 Args: 

203 dec: 

204 Declination of each source in radians relative 

205 to the celestial north 

206 psi: 

207 Angle in radians between each source's 

208 natural polarization basis and the basis 

209 which has the 0th unit vector pointing along 

210 the celestial equator 

211 phi: 

212 Angle in radians between each source's right 

213 ascension and the right ascension of the 

214 geocenter 

215 detector_tensors: 

216 Detector tensor for each of the interferometers 

217 for which observed strain is being calculated, 

218 stacked along the 0th axis 

219 detector_vertices: 

220 Vertices for each interferometer's spatial location 

221 relative to the geocenter. Used to compute delay 

222 between the waveform observed at the geocenter and 

223 the one observed at the detector site. To avoid 

224 adding any delay between the two, reset your coordinates 

225 such that the desired interferometer is at `(0., 0., 0.)`. 

226 sample_rate: 

227 Rate at which the polarization timeseries have been sampled 

228 polarziations: 

229 Timeseries for each waveform polarization which 

230 contributes to the interferometer response. Allowed 

231 polarizations are `cross`, `plus`, and `breathing`. 

232 Returns: 

233 Tensor representing the observed strain at each 

234 interferometer for each waveform. 

235 """ 

236 

237 # TODO: just use theta as the input parameter? 

238 # note that ** syntax is ordered, so we're safe 

239 # to be lazy and use `list` for the keys and values 

240 theta = torch.pi / 2 - dec 

241 antenna_responses = compute_antenna_responses( 

242 theta, psi, phi, detector_tensors, list(polarizations) 

243 ) 

244 

245 polarizations = torch.stack(list(polarizations.values()), axis=1) 

246 waveforms = torch.einsum( 

247 "...pi,...pt->...it", antenna_responses, polarizations 

248 ) 

249 

250 return shift_responses( 

251 waveforms, theta, phi, detector_vertices, sample_rate 

252 )

255def get_ifo_geometry( 

256 *ifos: str, 

257) -> Tuple[NetworkDetectorTensors, NetworkVertices]: 

258 """ 

259 For a given list of interferometer names, retrieve and 

260 concatenate the associated detector tensors and vertices 

261 of those interferometers. 

262 

263 Args: 

264 ifos: Names of the interferometers whose geometry to retrieve 

265 Returns: 

266 A concatenation of the detector tensors of each interferometer 

267 A concatenation of the vertices of each interferometer 

268 """ 

269 

270 tensors, vertices = [], [] 

271 for ifo in ifos: 

272 ifo = InterferometerGeometry(ifo) 

273 detector_tensor = plus(ifo.x_arm, ifo.y_arm) / 2 

274 tensors.append(detector_tensor) 

275 vertices.append(ifo.vertex) 

276 

277 tensors = torch.stack(tensors) 

278 vertices = torch.stack(vertices) 

279 return torch.Tensor(tensors), torch.Tensor(vertices)

282def compute_ifo_snr( 

283 responses: WaveformTensor, 

284 psd: PSDTensor, 

285 sample_rate: float, 

286 highpass: Union[float, Float[Tensor, " frequency"], None] = None, 

287 lowpass: Union[float, Float[Tensor, " frequency"], None] = None, 

288) -> Float[Tensor, "batch num_ifos"]: 

289 r"""Compute the SNRs of a batch of interferometer responses 

290 

291 Compute the signal to noise ratio (SNR) of individual 

292 interferometer responses to gravitational waveforms with 

293 respect to a background PSD for each interferometer. The 

294 SNR of the $i$th waveform at the $j$th interferometer 

295 is computed as: 

296 

297 $$\rho_{ij} = 

298 4 \int_{f_{\text{min}}}^{f_{\text{max}}} 

299 \frac{\tilde{h_{ij}}(f)\tilde{h_{ij}}^*(f)} 

300 {S_n^{(j)}(f)}df$$ 

301 

302 Where $f_{\text{min}}$ is a minimum frequency denoted 

303 by `highpass`, `f_{\text{max}}` is the maximum frequency 

304 denoted by `lowpass`, which defaults to the Nyquist frequency 

305 dictated by `sample_rate`; `\tilde{h_{ij}}` and `\tilde{h_{ij}}*` 

306 indicate the fourier transform of the $i$th waveform at 

307 the $j$th inteferometer and its complex conjugate, respectively; 

308 and $S_n^{(j)}$ is the backround PSD at the $j$th interferometer. 

309 

310 Args: 

311 responses: 

312 A batch of interferometer responses to a batch of 

313 raw gravitational waveforms 

314 psd: 

315 The one-sided power spectral density of the background 

316 noise at each interferometer to which a response 

317 in `responses` has been calculated. If 2D, each row of 

318 `psd` will be assumed to be the background PSD for each 

319 channel of _every_ batch element in `responses`. If 3D, 

320 this should contain a background PSD for each channel 

321 of each element in `responses`, and therefore the first 

322 two dimensions of `psd` and `responses` should match. 

323 sample_rate: 

324 The frequency at which the waveform responses timeseries 

325 have been sampled. Upon fourier transforming, should 

326 match the frequency resolution of the provided PSDs. 

327 highpass: 

328 The minimum frequency above which to compute the SNR. 

329 If a tensor is provided, it will be assumed to be a 

330 pre-computed mask used to 0-out low frequency components. 

331 If a float, it will be used to compute such a mask. If 

332 left as `None`, all frequencies up to `lowpass` 

333 will contribute to the SNR calculation. 

334 lowpass: 

335 The maximum frequency below which to compute the SNR. 

336 If a tensor is provided, it will be assumed to be a 

337 pre-computed mask used to 0-out high frequency components. 

338 If a float, it will be used to compute such a mask. If 

339 left as `None`, all frequencies from `highpass` up to 

340 the Nyquist freqyency will contribute to the SNR calculation. 

341 Returns: 

342 Batch of SNRs computed for each interferometer 

343 """ 

344 

345 # TODO: should we do windowing here? 

346 # compute frequency power, upsampling precision so that 

347 # computing absolute value doesn't accidentally zero some 

348 # values out. 

349 fft = torch.fft.rfft(responses, axis=-1).type(torch.complex128) 

350 fft = fft.abs() / sample_rate 

351 

352 # divide by background asd, then go back to FP32 precision 

353 # and square now that values are back in a reasonable range 

354 integrand = fft / (psd**0.5) 

355 integrand = integrand.type(torch.float32) ** 2 

356 

357 # mask out frequency components if a critical 

358 # frequency or frequency mask was provided 

359 if highpass is not None: 

360 if not isinstance(highpass, torch.Tensor): 

361 freqs = torch.fft.rfftfreq(responses.shape[-1], 1 / sample_rate) 

362 highpass = freqs >= highpass 

363 elif len(highpass) != integrand.shape[-1]: 

364 raise ValueError( 

365 "Can't apply highpass filter mask with {} frequency bins" 

366 "to signal fft with {} frequency bins".format( 

367 len(highpass), integrand.shape[-1] 

368 ) 

369 ) 

370 integrand *= highpass.to(integrand.device) 

371 if lowpass is not None: 

372 if not isinstance(lowpass, torch.Tensor): 

373 freqs = torch.fft.rfftfreq(responses.shape[-1], 1 / sample_rate) 

374 lowpass = freqs < lowpass 

375 elif len(lowpass) != integrand.shape[-1]: 

376 raise ValueError( 

377 "Can't apply lowpass filter mask with {} frequency bins" 

378 "to signal fft with {} frequency bins".format( 

379 len(lowpass), integrand.shape[-1] 

380 ) 

381 ) 

382 integrand *= lowpass.to(integrand.device) 

383 

384 # sum over the desired frequency range and multiply 

385 # by df to turn it into an integration (and get 

386 # our units to drop out) 

387 # TODO: we could in principle do this without requiring 

388 # that the user specify the sample rate by taking the 

389 # fft as-is (without dividing by sample rate) and then 

390 # taking the mean here (or taking the sum and dividing 

391 # by the sum of `highpass` if it's a mask). If we want 

392 # to allow the user to pass a float for highpass, we'll 

393 # need the sample rate to compute the mask, but if we 

394 # replace this with a `mask` argument instead we're in 

395 # the clear 

396 df = sample_rate / responses.shape[-1] 

397 integrated = integrand.sum(axis=-1) * df 

398 

399 # multiply by 4 for mystical reasons 

400 integrated = 4 * integrated # rho-squared 

401 return torch.sqrt(integrated)

404def compute_network_snr( 

405 responses: WaveformTensor, 

406 psd: PSDTensor, 

407 sample_rate: float, 

408 highpass: Union[float, Float[Tensor, " frequency"], None] = None, 

409 lowpass: Union[float, Float[Tensor, " frequency"], None] = None, 

410) -> BatchTensor: 

411 r""" 

412 Compute the total SNR from a gravitational waveform 

413 from a network of interferometers. The total SNR for 

414 the $i$th waveform is computed as 

415 

416 $$\rho_i = \sqrt{\sum_{j}^{N}\rho_{ij}^2}$$ 

417 

418 where \rho_{ij} is the SNR for the $i$th waveform at 

419 the $j$th interferometer in the network and $N$ is 

420 the total number of interferometers. 

421 

422 Args: 

423 responses: 

424 A batch of interferometer responses to a batch of 

425 raw gravitational waveforms 

426 backgrounds: 

427 The one-sided power spectral density of the background 

428 noise at each interferometer to which a response 

429 in `responses` has been calculated. If 2D, each row of 

430 `psd` will be assumed to be the background PSD for each 

431 channel of _every_ batch element in `responses`. If 3D, 

432 this should contain a background PSD for each channel 

433 of each element in `responses`, and therefore the first 

434 two dimensions of `psd` and `responses` should match. 

435 sample_rate: 

436 The frequency at which the waveform responses timeseries 

437 have been sampled. Upon fourier transforming, should 

438 match the frequency resolution of the provided PSDs. 

439 highpass: 

440 The minimum frequency above which to compute the SNR. 

441 If a tensor is provided, it will be assumed to be a 

442 pre-computed mask used to 0-out low frequency components. 

443 If a float, it will be used to compute such a mask. If 

444 left as `None`, all frequencies up to `sample_rate / 2` 

445 will contribute to the SNR calculation. 

446 lowpass: 

447 The maximum frequency below which to compute the SNR. 

448 If a tensor is provided, it will be assumed to be a 

449 pre-computed mask used to 0-out high frequency components. 

450 If a float, it will be used to compute such a mask. If 

451 left as `None`, all frequencies from `highpass` up to 

452 the Nyquist freqyency will contribute to the SNR calculation. 

453 Returns: 

454 Batch of SNRs for each waveform across the interferometer network 

455 """ 

456 snrs = compute_ifo_snr(responses, psd, sample_rate, highpass, lowpass) 

457 snrs = snrs**2 

458 return snrs.sum(axis=-1) ** 0.5

461def reweight_snrs( 

462 responses: WaveformTensor, 

463 target_snrs: Union[float, BatchTensor], 

464 psd: PSDTensor, 

465 sample_rate: float, 

466 highpass: Union[float, Float[Tensor, " frequency"], None] = None, 

467 lowpass: Union[float, Float[Tensor, " frequency"], None] = None, 

468) -> WaveformTensor: 

469 """Scale interferometer responses such that they have a desired SNR 

470 

471 Args: 

472 responses: 

473 A batch of interferometer responses to a batch of 

474 raw gravitational waveforms 

475 target_snrs: 

476 Either a tensor of desired SNRs for each waveform, 

477 or a single SNR to which all waveforms should be scaled. 

478 psd: 

479 The one-sided power spectral density of the background 

480 noise at each interferometer to which a response 

481 in `responses` has been calculated. If 2D, each row of 

482 `psd` will be assumed to be the background PSD for each 

483 channel of _every_ batch element in `responses`. If 3D, 

484 this should contain a background PSD for each channel 

485 of each element in `responses`, and therefore the first 

486 two dimensions of `psd` and `responses` should match. 

487 sample_rate: 

488 The frequency at which the waveform responses timeseries 

489 have been sampled. Upon fourier transforming, should 

490 match the frequency resolution of the provided PSDs. 

491 highpass: 

492 The minimum frequency above which to compute the SNR. 

493 If a tensor is provided, it will be assumed to be a 

494 pre-computed mask used to 0-out low frequency components. 

495 If a float, it will be used to compute such a mask. If 

496 left as `None`, all frequencies up to `sample_rate / 2` 

497 will contribute to the SNR calculation. 

498 lowpass: 

499 The maximum frequency below which to compute the SNR. 

500 If a tensor is provided, it will be assumed to be a 

501 pre-computed mask used to 0-out high frequency components. 

502 If a float, it will be used to compute such a mask. If 

503 left as `None`, all frequencies from `highpass` up to 

504 the Nyquist freqyency will contribute to the SNR calculation. 

505 Returns: 

506 Rescaled interferometer responses 

507 """ 

508 

509 snrs = compute_network_snr(responses, psd, sample_rate, highpass, lowpass) 

510 weights = target_snrs / snrs 

511 return responses * weights[:, None, None]
